995 void Graphics3D::_rotateZtoDirection(const Vec3<float> &direction) {
996  float len = direction.norm();
997  float dxn = direction(0) / len;
998  float dyn = direction(1) / len;
999  float dzn = direction(2) / len;
1000 
1001  // Note: We need to create a rotation such that the z-axis points in the
1002  // direction of the 'direction' argument
1003  // Thus, we can realize the rotation with a pure x-y axial rotation
1004  // The rotation matrix of the rotated frame 'r' to the current frame 'c'
1005  // (c_R_r) has special form. It's 3-rd column in particular has form: [
1006  // wy*sin(theta) -wx*sin(theta) (1- cos(theta))]' where theta , [wx wy
1007  // 0] is the angle, axis of rotation
1008 
1009  // Find the rotation angle by comparing the z-axis of the current and rotated
1010  // frame
1011  const double cosTheta = dzn;
1012  const double theta = acos(cosTheta);
1013  const double sinTheta = sin(theta);
1014 
1015  // Exploit the special form of the rotation matrix (explained above) for find
1016  // the axis of rotation
1017  float rX, rY, rZ;
1018  if (theta > 0) {
1019  rX = -dyn / sinTheta;
1020  rY = dxn / sinTheta;
1021  rZ = 0;
1022  } else {
1023  rX = 0;
1024  rY = 0;
1025  rZ = 1;
1026  }
1027  glRotatef(ori::rad2deg(theta), rX, rY, rZ);
1028 }
