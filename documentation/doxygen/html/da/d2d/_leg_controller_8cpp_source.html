264 template <typename T>
265 void computeLegJacobianAndPosition(Quadruped<T>& quad, Vec3<T>& q, Mat3<T>* J,
266  Vec3<T>* p, int leg) {
267  T l1 = quad._abadLinkLength;
268  T l2 = quad._hipLinkLength;
269  T l3 = quad._kneeLinkLength;
270  T l4 = quad._kneeLinkY_offset;
271  T sideSign = quad.getSideSign(leg);
272 
273  T s1 = std::sin(q(0));
274  T s2 = std::sin(q(1));
275  T s3 = std::sin(q(2));
276 
277  T c1 = std::cos(q(0));
278  T c2 = std::cos(q(1));
279  T c3 = std::cos(q(2));
280 
281  T c23 = c2 * c3 - s2 * s3;
282  T s23 = s2 * c3 + c2 * s3;
283 
284  if (J) {
285  J->operator()(0, 0) = 0;
286  J->operator()(0, 1) = l3 * c23 + l2 * c2;
287  J->operator()(0, 2) = l3 * c23;
288  J->operator()(1, 0) = l3 * c1 * c23 + l2 * c1 * c2 - (l1+l4) * sideSign * s1;
289  J->operator()(1, 1) = -l3 * s1 * s23 - l2 * s1 * s2;
290  J->operator()(1, 2) = -l3 * s1 * s23;
291  J->operator()(2, 0) = l3 * s1 * c23 + l2 * c2 * s1 + (l1+l4) * sideSign * c1;
292  J->operator()(2, 1) = l3 * c1 * s23 + l2 * c1 * s2;
293  J->operator()(2, 2) = l3 * c1 * s23;
294  }
295 
296  if (p) {
297  p->operator()(0) = l3 * s23 + l2 * s2;
298  p->operator()(1) = (l1+l4) * sideSign * c1 + l3 * (s1 * c23) + l2 * c2 * s1;
299  p->operator()(2) = (l1+l4) * sideSign * s1 - l3 * (c1 * c23) - l2 * c1 * c2;
300  }
301 }
