132  "Must have 3x3 matrix");
133  Quat<typename T::Scalar> q;
134  Mat3<typename T::Scalar> r = r1.transpose();
135  typename T::Scalar tr = r.trace();
136  if (tr > 0.0) {
137  typename T::Scalar S = sqrt(tr + 1.0) * 2.0;
138  q(0) = 0.25 * S;
139  q(1) = (r(2, 1) - r(1, 2)) / S;
140  q(2) = (r(0, 2) - r(2, 0)) / S;
141  q(3) = (r(1, 0) - r(0, 1)) / S;
142  } else if ((r(0, 0) > r(1, 1)) && (r(0, 0) > r(2, 2))) {
143  typename T::Scalar S = sqrt(1.0 + r(0, 0) - r(1, 1) - r(2, 2)) * 2.0;
144  q(0) = (r(2, 1) - r(1, 2)) / S;
145  q(1) = 0.25 * S;
146  q(2) = (r(0, 1) + r(1, 0)) / S;
147  q(3) = (r(0, 2) + r(2, 0)) / S;
148  } else if (r(1, 1) > r(2, 2)) {
149  typename T::Scalar S = sqrt(1.0 + r(1, 1) - r(0, 0) - r(2, 2)) * 2.0;
150  q(0) = (r(0, 2) - r(2, 0)) / S;
151  q(1) = (r(0, 1) + r(1, 0)) / S;
152  q(2) = 0.25 * S;
153  q(3) = (r(1, 2) + r(2, 1)) / S;

157  q(1) = (r(0, 2) + r(2, 0)) / S;
