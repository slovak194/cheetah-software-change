242 void ControlParameters::initializeFromIniFile(const std::string& path) {
243  INIReader iniReader(path);
244  if (iniReader.ParseError() < 0) {
245  printf(

253 
254  if (sections.size() != 1) {
255  printf(
256  "[ERROR] INI file %s had %ld sections (expected 1) : not initializing "
257  "control parameters\n",

259  throw std::runtime_error("ini file bad");
260  }
261 
262  std::string sectionName = *(sections.begin());
263 
264  if (sectionName != _name) {
265  printf(
266  "[ERROR] INI file %s has section name %s, which cannot be used to "
267  "initialize %s\n",
268  path.c_str(), sectionName.c_str(), _name.c_str());
269  throw std::runtime_error("ini file bad");
270  }
271

273 
274  for (auto& name : parameterNames) {

276  switch (cp._kind) {
277  case ControlParameterValueKind::DOUBLE:
278  cp.initializeDouble(iniReader.GetReal(sectionName, name, 0.));
279  break;
280  case ControlParameterValueKind::FLOAT:
281  cp.initializeFloat((float)iniReader.GetReal(sectionName, name, 0.));
282  break;
