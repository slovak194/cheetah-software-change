488 void Simulation::buildLcmMessage() {
489  _simLCM.time = _currentSimTime;
490  _simLCM.timesteps = _highLevelIterations;
491  auto& state = _simulator->getState();
492  auto& dstate = _simulator->getDState();
493 
494  Vec3<double> rpy = ori::quatToRPY(state.bodyOrientation);
495  RotMat<double> Rbody = ori::quaternionToRotationMatrix(state.bodyOrientation);
496  Vec3<double> omega = Rbody.transpose() * state.bodyVelocity.head<3>();
497  Vec3<double> v = Rbody.transpose() * state.bodyVelocity.tail<3>();
498 
499  for (size_t i = 0; i < 4; i++) {
500  _simLCM.quat[i] = state.bodyOrientation[i];
501  }
502 
503  for (size_t i = 0; i < 3; i++) {
504  _simLCM.vb[i] = state.bodyVelocity[i + 3]; // linear velocity in body frame
505  _simLCM.rpy[i] = rpy[i];
506  for (size_t j = 0; j < 3; j++) {
507  _simLCM.R[i][j] = Rbody(i, j);
508  }
509  _simLCM.omegab[i] = state.bodyVelocity[i];
510  _simLCM.omega[i] = omega[i];
511  _simLCM.p[i] = state.bodyPosition[i];
512  _simLCM.v[i] = v[i];
513  _simLCM.vbd[i] = dstate.dBodyVelocity[i + 3];
514  }
515 
516  for (size_t leg = 0; leg < 4; leg++) {
517  for (size_t joint = 0; joint < 3; joint++) {
518  _simLCM.q[leg][joint] = state.q[leg * 3 + joint];
519  _simLCM.qd[leg][joint] = state.qd[leg * 3 + joint];
520  _simLCM.qdd[leg][joint] = dstate.qdd[leg * 3 + joint];
521  _simLCM.tau[leg][joint] = _tau[leg * 3 + joint];
522  size_t gcID = _simulator->getModel()._footIndicesGC.at(leg);
523  _simLCM.p_foot[leg][joint] = _simulator->getModel()._pGC.at(gcID)[joint];
524  _simLCM.f_foot[leg][joint] = _simulator->getContactForce(gcID)[joint];
525  }

658 void Simulation::loadTerrainFile(const std::string& terrainFileName,
659  bool addGraphics) {
660  printf("load terrain %s\n", terrainFileName.c_str());
661  ParamHandler paramHandler(terrainFileName);
662 
663  if (!paramHandler.fileOpenedSuccessfully()) {
664  printf("[ERROR] could not open yaml file for terrain\n");
665  throw std::runtime_error("yaml bad");
666  }
667 
668  std::vector<std::string> keys = paramHandler.getKeys();
669 
670  for (auto& key : keys) {
671  auto load = [&](double& val, const std::string& name) {
672  if (!paramHandler.getValue<double>(key, name, val))
673  throw std::runtime_error("terrain read bad: " + key + " " + name);
674  };
675 
676  auto loadVec = [&](double& val, const std::string& name, size_t idx) {
677  std::vector<double> v;
678  if (!paramHandler.getVector<double>(key, name, v))
679  throw std::runtime_error("terrain read bad: " + key + " " + name);
680  val = v.at(idx);
681  };
682 
683  auto loadArray = [&](double* val, const std::string& name, size_t idx) {
684  std::vector<double> v;
685  if (!paramHandler.getVector<double>(key, name, v))
686  throw std::runtime_error("terrain read bad: " + key + " " + name);
687  assert(v.size() == idx);
688  for (size_t i = 0; i < idx; i++) val[i] = v[i];
689  };
690 
691  printf("terrain element %s\n", key.c_str());
692  std::string typeName;
693  paramHandler.getString(key, "type", typeName);
694  if (typeName == "infinite-plane") {
695  double mu, resti, height, gfxX, gfxY, checkerX, checkerY;
696  load(mu, "mu");
697  load(resti, "restitution");
698  load(height, "height");
699  loadVec(gfxX, "graphicsSize", 0);
700  loadVec(gfxY, "graphicsSize", 1);
701  loadVec(checkerX, "checkers", 0);
702  loadVec(checkerY, "checkers", 1);
703  addCollisionPlane(mu, resti, height, gfxX, gfxY, checkerX, checkerY,
704  addGraphics);
705  } else if (typeName == "box") {
706  double mu, resti, depth, width, height, transparent;
707  double pos[3];
708  double ori[3];
709  load(mu, "mu");
710  load(resti, "restitution");
711  load(depth, "depth");
712  load(width, "width");
713  load(height, "height");
714  loadArray(pos, "position", 3);
715  loadArray(ori, "orientation", 3);
716  load(transparent, "transparent");
717 
718  Mat3<double> R_box = ori::rpyToRotMat(Vec3<double>(ori));
719  R_box.transposeInPlace(); // collisionBox uses "rotation" matrix instead
720  // of "transformation"
721  addCollisionBox(mu, resti, depth, width, height, Vec3<double>(pos), R_box,
722  addGraphics, transparent != 0.);
723  } else if (typeName == "stairs") {
724  double mu, resti, rise, run, stepsDouble, width, transparent;
725  double pos[3];
726  double ori[3];
727  load(mu, "mu");
728  load(resti, "restitution");
729  load(rise, "rise");
730  load(width, "width");
731  load(run, "run");
732  load(stepsDouble, "steps");
733  loadArray(pos, "position", 3);
734  loadArray(ori, "orientation", 3);
735  load(transparent, "transparent");
736 
737  Mat3<double> R = ori::rpyToRotMat(Vec3<double>(ori));
738  Vec3<double> pOff(pos);
739  R.transposeInPlace(); // "graphics" rotation matrix
740 
741  size_t steps = (size_t)stepsDouble;
742 
743  double heightOffset = rise / 2;
744  double runOffset = run / 2;
745  for (size_t step = 0; step < steps; step++) {
746  Vec3<double> p(runOffset, 0, heightOffset);
747  p = R * p + pOff;
748 
749  addCollisionBox(mu, resti, run, width, heightOffset * 2, p, R,
750  addGraphics, transparent != 0.);
751 
752  heightOffset += rise / 2;
753  runOffset += run;
754  }
755  } else if (typeName == "mesh") {
756  double mu, resti, transparent, grid;
757  Vec3<double> left_corner;
758  std::vector<std::vector<double> > height_map_2d;
759  load(mu, "mu");
760  load(resti, "restitution");
761  load(transparent, "transparent");
762  load(grid, "grid");
763  loadVec(left_corner[0], "left_corner_loc", 0);
764  loadVec(left_corner[1], "left_corner_loc", 1);
765  loadVec(left_corner[2], "left_corner_loc", 2);
766 
767  int x_len(0);
768  int y_len(0);
769  bool file_input(false);
770  paramHandler.getBoolean(key, "heightmap_file", file_input);
771  if (file_input) {
772  // Read from text file
773  std::string file_name;
774  paramHandler.getString(key, "heightmap_file_name", file_name);
775  std::ifstream f_height;
776  f_height.open(THIS_COM "/config/" + file_name);
777  if (!f_height.good()) {
778  std::cout << "file reading error: "
779  << THIS_COM "../config/" + file_name << std::endl;
780  }
781  int i(0);
782  int j(0);
783  double tmp;
784 
785  std::string line;
786  std::vector<double> height_map_vec;
787  while (getline(f_height, line)) {
788  std::istringstream iss(line);
789  j = 0;
790  while (iss >> tmp) {
791  height_map_vec.push_back(tmp);
792  ++j;
793  }
794  y_len = j;
795  height_map_2d.push_back(height_map_vec);
796  height_map_vec.clear();
797  // printf("y len: %d\n", y_len);
798  ++i;
799  }
800  x_len = i;
801 
802  } else {
803  paramHandler.get2DArray(key, "height_map", height_map_2d);
804  x_len = height_map_2d.size();
805  y_len = height_map_2d[0].size();
806  // printf("x, y len: %d, %d\n", x_len, y_len);
807  }
808 
809  DMat<double> height_map(x_len, y_len);
810  for (int i(0); i < x_len; ++i) {
811  for (int j(0); j < y_len; ++j) {
812  height_map(i, j) = height_map_2d[i][j];
813  // printf("height (%d, %d) : %f\n", i, j, height_map(i,j) );
814  }
815  }
816  addCollisionMesh(mu, resti, grid, left_corner, height_map, addGraphics,
817  transparent != 0.);
818 
819  } else {
820  throw std::runtime_error("unknown terrain " + typeName);
821  }
822  }
823 }
