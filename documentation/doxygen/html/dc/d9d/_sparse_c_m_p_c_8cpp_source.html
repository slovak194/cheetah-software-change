188  u32 constraint_idx = addConstraint(12);
189  for(u32 i = 0; i < 12; i++) { // diagonal of the identity
190  _constraintTriples.push_back({1,constraint_idx + i, state_idx + i});
191  }
192

194 
195  u32 ctrl_cnt = _contactCounts[0];
196 
197  for(u32 i = 0; i < ctrl_cnt; i++) { // Bblocks within this b (contact feet)
198  // select -B[0]*u[0]
199  u32 bbIdx = _runningContactCounts[0] + i;
200  for(u32 ax = 0; ax < 3; ax++) { // columns within the b block (forces axes)
201  for(u32 row = 0; row < 12; row++) { // rows within the b block
202  addConstraintTriple(-_bBlocks[bbIdx](row, ax), constraint_idx + row, getControlIndex(bbIdx) + ax);

347  //printf("[SparseCMPC] Run %d, %d\n", varCount, _constraintCount);
348  assert(_constraintCount == _ub.size());
349  assert(_constraintCount == _lb.size());
350  assert(varCount == _linearCost.size());
351 
352 
353  QpProblem<double> solver(varCount, _constraintCount, false);
354  solver.A_triples = std::move(_constraintTriples);
355  solver.P_triples = std::move(_costTriples);
356  solver.u = Vector<double>(_ub.size());
357  solver.l = Vector<double>(_ub.size());
358  for(u32 i = 0; i < _constraintCount; i++) {
359  solver.u[i] = _ub[i];
360  solver.l[i] = _lb[i];
361  }
362  solver.q = Vector<double>(varCount);
363  for(u32 i = 0; i < varCount; i++) {
364  solver.q[i] = _linearCost[i];
365  }
366 
367  solver.settings.alpha = 1.5;
368  solver.settings.rho = 1e-3;
369  solver.settings.terminate = 1e-6;
370  solver.settings.sigma = 1e-6;
371 
372 
373  solver.runFromTriples(-1, true);
374  _result = solver.getSolution().cast<float>();
