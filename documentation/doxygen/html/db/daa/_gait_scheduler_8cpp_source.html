99  gaitData.initialPhase = fmod((gaitData.initialPhase + (dt / gaitData.periodTimeNominal)), 1);

101 
102  // Iterate over the feet

104  // Set the previous contact state for the next timestep
105  gaitData.contactStatePrev(foot) = gaitData.contactStateScheduled(foot);
106 
107  if (gaitData.gaitEnabled(foot) == 1) {
108  // Monotonic time based phase incrementation
109  if (gaitData._currentGait == GaitType::STAND) {
110  // Don't increment the phase when in stand mode
111  dphase = 0.0;
112  } else {
113  dphase = gaitData.phaseScale(foot) * (dt / gaitData.periodTimeNominal);
114  }
115 
116  // Find each foot's current phase
117  gaitData.phaseVariable(foot) =
118  fmod((gaitData.phaseVariable(foot) + dphase), 1);
119 
120  // Check the current contact state
121  if (gaitData.phaseVariable(foot) <= gaitData.switchingPhase(foot)) {
122  // Foot is scheduled to be in contact
123  gaitData.contactStateScheduled(foot) = 1;
124 
125  // Stance subphase calculation
126  gaitData.phaseStance(foot) =
127  gaitData.phaseVariable(foot) / gaitData.switchingPhase(foot);
128 
129  // Swing phase has not started since foot is in stance
130  gaitData.phaseSwing(foot) = 0.0;
131 
132  // Calculate the remaining time in stance
133  gaitData.timeStanceRemaining(foot) =
134  gaitData.periodTime(foot) *
135  (gaitData.switchingPhase(foot) - gaitData.phaseVariable(foot));
136 
137  // Foot is in stance, no swing time remaining
138  gaitData.timeSwingRemaining(foot) = 0.0;
139 
140  // First contact signifies scheduled touchdown
141  if (gaitData.contactStatePrev(foot) == 0) {
142  // Set the touchdown flag to 1
143  gaitData.touchdownScheduled(foot) = 1;
144 
145  } else {
146  // Set the touchdown flag to 0
147  gaitData.touchdownScheduled(foot) = 0;
148  }
149 
150  } else {
151  // Foot is not scheduled to be in contact
152  gaitData.contactStateScheduled(foot) = 0;
153 
154  // Stance phase has completed since foot is in swing
155  gaitData.phaseStance(foot) = 1.0;
156 
157  // Swing subphase calculation
158  gaitData.phaseSwing(foot) =
159  (gaitData.phaseVariable(foot) - gaitData.switchingPhase(foot)) /
160  (1.0 - gaitData.switchingPhase(foot));
161

163  gaitData.timeStanceRemaining(foot) = 0.0;
164

166  gaitData.timeSwingRemaining(foot) =
167  gaitData.periodTime(foot) * (1 - gaitData.phaseVariable(foot));
168

171  // Set the liftoff flag to 1
172  gaitData.liftoffScheduled(foot) = 1;
173

176  gaitData.liftoffScheduled(foot) = 0;
177  }

179 
180  } else {

217  gaitData._nextGait = GaitType(userParameters->gait_type);
